    <script>
        class KingTrumbGame {
            constructor() {
                this.balance = 0;
                this.energy = 100;
                this.tapPower = 1;
                this.init();
            }

            init() {
                // Load saved data
                const saved = localStorage.getItem('kingTrumbData');
                if (saved) {
                    const data = JSON.parse(saved);
                    this.balance = data.balance || 0;
                    this.energy = data.energy || 100;
                    this.tapPower = data.tapPower || 1;
                }

                this.setupEventListeners();
                this.updateUI();
                this.startEnergyRecovery();
            }

            setupEventListeners() {
                document.getElementById('tapButton').addEventListener('click', () => this.tap());
            }

            tap() {
                if (this.energy > 0) {
                    this.balance += this.tapPower;
                    this.energy -= 1;
                    this.updateUI();
                    this.saveGame();
                    this.createTapEffect();
                }
            }

            updateUI() {
                document.getElementById('balance').textContent = this.balance + ' KTR';
                document.getElementById('energy').style.width = this.energy + '%';
                document.getElementById('energyText').textContent = this.energy + '/100';
            }

            createTapEffect() {
                const effect = document.createElement('div');
                effect.textContent = '+' + this.tapPower + ' KTR';
                effect.style.position = 'fixed';
                effect.style.color = '#FFD700';
                effect.style.fontWeight = 'bold';
                effect.style.pointerEvents = 'none';
                effect.style.animation = 'floatUp 1s forwards';
                effect.style.left = '50%';
                effect.style.top = '50%';
                effect.style.transform = 'translate(-50%, -50%)';
                
                document.body.appendChild(effect);
                setTimeout(() => effect.remove(), 1000);
            }

            startEnergyRecovery() {
                setInterval(() => {
                    if (this.energy < 100) {
                        this.energy += 1;
                        this.updateUI();
                        this.saveGame();
                    }
                }, 3000);
            }

            saveGame() {
                const data = {
                    balance: this.balance,
                    energy: this.energy,
                    tapPower: this.tapPower
                };
                localStorage.setItem('kingTrumbData', JSON.stringify(data));
            }
        }

        // Initialize game
        const game = new KingTrumbGame();

        // Button functions
        function withdraw() {
